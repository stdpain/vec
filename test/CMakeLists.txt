FUNCTION(ADD_BENCH TEST_NAME)
    ADD_EXECUTABLE(${TEST_NAME}.out ${CMAKE_CURRENT_SOURCE_DIR}/bench/${TEST_NAME}.cpp)
    TARGET_LINK_LIBRARIES(${TEST_NAME}.out vec benchmark pthread)
ENDFUNCTION()
